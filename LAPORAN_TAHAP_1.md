## 1. Deskripsi Aplikasi

**SpareHub** adalah aplikasi web e-commerce B2C yang dirancang untuk jual-beli suku cadang kendaraan (spare parts). Platform ini memungkinkan pengguna untuk berbelanja berbagai produk suku cadang kendaraan dengan mudah dan nyaman.

### Fitur Utama:

#### Untuk Pembeli/Customer:
- **Browsing Produk**: Melihat katalog produk suku cadang yang tersedia
- **Detail Produk**: Melihat informasi lengkap produk termasuk harga, diskon, dan rating
- **Keranjang Belanja**: Menambahkan produk ke keranjang dan mengatur jumlah
- **Checkout**: Proses pembelian dengan pemilihan alamat dan metode pembayaran
- **Rating & Review**: Memberikan penilaian dan komentar pada produk
- **Riwayat Pesanan**: Melihat daftar pesanan yang telah dilakukan
- **Manajemen Profil**: Mengubah informasi pribadi pengguna

#### Untuk Penjual/Seller:
- **Profil Toko**: Mengelola informasi toko (nama, lokasi, logo)
- **Manajemen Produk**: Menambah, mengubah, dan menghapus produk (CRUD)
- **Statistik Toko**: Melihat total produk dan informasi toko

#### Fitur Umum:
- **Autentikasi Pengguna**: Login dan Register dengan validasi
- **Navigasi Konsisten**: Navbar yang unified di semua halaman
- **Responsive Design**: Tampilan yang menyesuaikan berbagai ukuran layar
- **Persistensi Data**: Penyimpanan data menggunakan localStorage

### Teknologi yang Digunakan:
- **HTML5**: Struktur halaman web
- **CSS3**: Styling dan responsive design
- **JavaScript (ES6+)**: Logika aplikasi dan interaksi
- **localStorage**: Penyimpanan data di browser
- **JSON**: Format data untuk produk, user, dan transaksi
- **Font Awesome**: Icon library untuk UI

### Arsitektur Aplikasi:
Aplikasi ini dibangun sebagai **frontend-only application** tanpa backend server. Data disimpan dalam file JSON dan localStorage browser untuk persistensi data runtime.

```javascript
    const email = document.getElementById('email').value;

    const users = await response.json();

    const user = users.find(u => u.email === email && u.password === password);
```

```javascript
let cartData = [];
let selectedAddress = null;
let selectedPayment = null;

async function loadCheckoutData() {
    // Load cart from localStorage
    cartData = JSON.parse(localStorage.getItem('keranjangData')) || [];

    // Display cart items
    displayOrderItems();

    // Calculate prices
    calculatePrices();

    // Load addresses
    await loadAddresses();
}

function displayOrderItems() {
    const container = document.getElementById('orderItems');
    container.innerHTML = cartData.map(item => `
        <div class="order-item">
            <img src="${item.image}" alt="${item.name}">
            <div class="item-details">
                <h4>${item.name}</h4>
                <p>Jumlah: ${item.quantity}</p>
            </div>
            <div class="item-price">
                Rp ${(item.price * item.quantity).toLocaleString()}
            </div>
        </div>
    `).join('');
}

function calculatePrices() {
    let subtotal = 0;
    let discount = 0;

    cartData.forEach(item => {
        const itemTotal = item.price * item.quantity;
        subtotal += itemTotal;

        if (item.discount) {
            discount += itemTotal * (item.discount / 100);
        }
    });

    const shipping = 0; // Free shipping
    const total = subtotal - discount + shipping;

    document.getElementById('subtotal').textContent = `Rp ${subtotal.toLocaleString()}`;
    document.getElementById('discount').textContent = `Rp ${discount.toLocaleString()}`;
    document.getElementById('total').textContent = `Rp ${total.toLocaleString()}`;
}

async function loadAddresses() {
    const response = await fetch('JSON/alamatData.json');
    const addresses = await response.json();

    const select = document.getElementById('addressSelect');
    addresses.forEach(addr => {
        const option = document.createElement('option');
        option.value = addr.id;
        option.textContent = addr.label;
        select.appendChild(option);
    });
}

// Handle address selection
document.getElementById('addressSelect').addEventListener('change', function() {

// Handle payment method selection
document.querySelectorAll('.payment-option').forEach(option => {

        document.querySelectorAll('.payment-option').forEach(opt => {

});

// Confirm payment
document.getElementById('confirmPayment').addEventListener('click', function() {

    // Create order
    const order = {

    let orders = JSON.parse(localStorage.getItem('orderData')) || [];
    orders.push(order);
```

```javascript
let cartData = [];

function loadCart() {
    const cartJSON = localStorage.getItem('keranjangData');
    cartData = cartJSON ? JSON.parse(cartJSON) : [];

    if (cartData.length === 0) {
        showEmptyCart();
    } else {
        hideEmptyCart();
        renderCartItems();
        calculateTotals();
    }
}

function renderCartItems() {
    const container = document.getElementById('cartItems');
    container.innerHTML = cartData.map((item, index) => `
        <div class="cart-item" data-index="${index}">
            <img src="${item.image}" alt="${item.name}" class="item-image">
            <div class="item-details">
                <h3 class="item-name">${item.name}</h3>
                <p class="item-price">Rp ${item.price.toLocaleString()}</p>
                ${item.discount ? `<span class="discount-badge">${item.discount}% OFF</span>` : ''}
            </div>
            <div class="item-quantity">
                <button class="qty-btn" onclick="updateQuantity(${index}, -1)">-</button>
                <input type="number" class="qty-input" value="${item.quantity}"
                       onchange="setQuantity(${index}, this.value)" min="1">
                <button class="qty-btn" onclick="updateQuantity(${index}, 1)">+</button>
            </div>
            <div class="item-subtotal">
                Rp ${(item.price * item.quantity).toLocaleString()}
            </div>
            <button class="btn-remove" onclick="removeItem(${index})">
                <i class="fas fa-trash"></i>
            </button>
        </div>
    `).join('');
}

function updateQuantity(index, change) {
    cartData[index].quantity += change;

    if (cartData[index].quantity < 1) {
        cartData[index].quantity = 1;
    }

    saveCart();
    renderCartItems();
    calculateTotals();
    updateCartBadge();
}

function setQuantity(index, value) {
    const qty = parseInt(value);
    if (qty >= 1) {
        cartData[index].quantity = qty;
        saveCart();
        calculateTotals();
        updateCartBadge();
    }
}

function removeItem(index) {
    if (confirm('Hapus produk ini dari keranjang?')) {
        cartData.splice(index, 1);
        saveCart();
        loadCart();
        updateCartBadge();
    }
}

function calculateTotals() {
    let subtotal = 0;
    let totalDiscount = 0;

    cartData.forEach(item => {
        const itemSubtotal = item.price * item.quantity;
        subtotal += itemSubtotal;

        if (item.discount) {
            totalDiscount += itemSubtotal * (item.discount / 100);
        }
    });

    const total = subtotal - totalDiscount;

    document.getElementById('subtotal').textContent = `Rp ${subtotal.toLocaleString()}`;
    document.getElementById('discount').textContent = `Rp ${totalDiscount.toLocaleString()}`;
    document.getElementById('total').textContent = `Rp ${total.toLocaleString()}`;
}

function saveCart() {
    localStorage.setItem('keranjangData', JSON.stringify(cartData));
}

function showEmptyCart() {
    document.getElementById('cartItems').style.display = 'none';
    document.querySelector('.cart-summary').style.display = 'none';
    document.getElementById('emptyCart').style.display = 'block';
}

function hideEmptyCart() {
    document.getElementById('cartItems').style.display = 'block';
    document.querySelector('.cart-summary').style.display = 'block';
    document.getElementById('emptyCart').style.display = 'none';
}

function updateCartBadge() {
    const totalItems = cartData.reduce((sum, item) => sum + item.quantity, 0);
    // Update navbar badge if NavbarManager exists
    if (window.navbarManager) {
        window.navbarManager.updateCartCount();
    }
}

function checkout() {
    if (cartData.length === 0) {
        alert('Keranjang belanja kosong!');
        return;
    }
    window.location.href = 'checkout.html';
}

// Initialize on page load
window.onload = loadCart;
```

```javascript
document.getElementById('registerForm').addEventListener('submit', async function(e) {
    e.preventDefault();

    // Get form values
    const username = document.getElementById('username').value.trim();
    const email = document.getElementById('email').value.trim();
    const password = document.getElementById('password').value;
    const confirmPassword = document.getElementById('confirmPassword').value;
    const phone = document.getElementById('phone').value.trim();
    const address = document.getElementById('address').value.trim();

    // Validation
    if (!validateForm(username, email, password, confirmPassword)) {
        return;
    }

    // Check if email already exists
    const existingUsers = await loadUsers();
    if (existingUsers.some(user => user.email === email)) {
        alert('Email sudah terdaftar! Gunakan email lain.');
        return;
    }

    // Create new user object
    const newUser = {
        id: Date.now(),
        username: username,
        email: email,
        password: password, // In production, should be hashed!
        phone: phone || '',
        address: address || '',
        createdAt: new Date().toISOString()
    };

    // Save user
    existingUsers.push(newUser);
    localStorage.setItem('userData', JSON.stringify(existingUsers));

    // Success message and redirect
    alert('Registrasi berhasil! Silakan login.');
    window.location.href = 'login.html';
});

function validateForm(username, email, password, confirmPassword) {
    // Username validation
    if (username.length < 3) {
        alert('Username minimal 3 karakter!');
        return false;
    }

    // Email validation
    const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
    if (!emailRegex.test(email)) {
        alert('Format email tidak valid!');
        return false;
    }

    // Password validation
    if (password.length < 6) {
        alert('Password minimal 6 karakter!');
        return false;
    }

    // Confirm password validation
    if (password !== confirmPassword) {
        alert('Password dan konfirmasi password tidak cocok!');
        return false;
    }

    return true;
}

async function loadUsers() {
    try {
        // Try to load from localStorage first
        const localData = localStorage.getItem('userData');
        if (localData) {
            return JSON.parse(localData);
        }

        // If not in localStorage, load from JSON file
        const response = await fetch('JSON/userData.json');
        const users = await response.json();
        return users;
    } catch (error) {
        console.error('Error loading users:', error);
        return [];
    }
}

function togglePassword(fieldId) {
    const field = document.getElementById(fieldId);
    const button = field.parentElement.querySelector('.toggle-password i');

    if (field.type === 'password') {
        field.type = 'text';
        button.classList.remove('fa-eye');
        button.classList.add('fa-eye-slash');
    } else {
        field.type = 'password';
        button.classList.remove('fa-eye-slash');
        button.classList.add('fa-eye');
    }
}
```

```javascript
async function loadProductDetail() {
    // Get product ID from URL
    const urlParams = new URLSearchParams(window.location.search);
    const productId = parseInt(urlParams.get('id'));

    if (!productId) {
        alert('Produk tidak ditemukan!');

function displayProductDetails() {
    const product = currentProduct;

    // Set image
    document.getElementById('productImage').src = product.image;
    document.getElementById('productImage').alt = product.name;

    // Set title
    document.getElementById('productName').textContent = product.name;

    // Set price
    document.getElementById('currentPrice').textContent = product.price.toLocaleString();

    // Set stock
    document.getElementById('stockCount').textContent = product.stock || 100;

    // Set store info
    document.getElementById('storeName').textContent = product.storeName || 'Nama Toko';
    document.getElementById('storeName').href = `profil_toko.html?id=${product.storeId}`;

    // Display rating
    displayRating(product.rating || 0);

function displayRating(rating) {
    const starsContainer = document.getElementById('productStars');
    let starsHTML = '';

    for (let i = 1; i <= 5; i++) {
        if (i <= rating) {
            starsHTML += '<i class="fas fa-star"></i>';
        } else {
            starsHTML += '<i class="far fa-star empty"></i>';
        }
    }

    starsContainer.innerHTML = starsHTML;
}

async function loadProductReviews(productId) {
    try {
        // Load ratings from localStorage
        const ratingsJSON = localStorage.getItem('ratingList');
        let ratings = ratingsJSON ? JSON.parse(ratingsJSON) : [];

        // Filter ratings for this product
        const productRatings = ratings.filter(r => r.productId === productId);

        // Update rating count
        document.getElementById('ratingCount').textContent = `(${productRatings.length} rating)`;

        // Display reviews
        displayReviews(productRatings);

        // Calculate and update average rating
        if (productRatings.length > 0) {
            const avgRating = productRatings.reduce((sum, r) => sum + r.rating, 0) / productRatings.length;
            displayRating(Math.round(avgRating));
        }
    } catch (error) {
        console.error('Error loading reviews:', error);
    }
}

function displayReviews(reviews) {
    const container = document.getElementById('reviewsList');

    if (reviews.length === 0) {
        container.innerHTML = '<p>Belum ada ulasan untuk produk ini.</p>';
        return;
    }

    container.innerHTML = reviews.map(review => `
        <div class="review-item">
            <div class="review-header">
                <span class="reviewer-name">${review.username}</span>
                <span class="review-date">${formatDate(review.timestamp)}</span>
            </div>
            <div class="review-stars">
                ${'★'.repeat(review.rating)}${'☆'.repeat(5 - review.rating)}
            </div>
            <p class="review-comment">${review.comment}</p>
        </div>
    `).join('');
}

function formatDate(timestamp) {
    const date = new Date(timestamp);
    return date.toLocaleDateString('id-ID', {
        year: 'numeric',
        month: 'long',
        day: 'numeric'
    });
}

function addToCart() {
    if (!currentProduct) return;

    // Get cart from localStorage
    let cart = JSON.parse(localStorage.getItem('keranjangData')) || [];

    // Check if product already in cart
    const existingIndex = cart.findIndex(item => item.id === currentProduct.id);

    if (existingIndex !== -1) {
        // Update quantity
        cart[existingIndex].quantity += quantity;
    } else {
        // Add new item
        cart.push({
            id: currentProduct.id,
            name: currentProduct.name,
            price: currentProduct.price,
            image: currentProduct.image,
            discount: currentProduct.discount || 0,
            quantity: quantity
        });
    }

    // Save cart
    localStorage.setItem('keranjangData', JSON.stringify(cart));

    // Update cart badge if navbar manager exists
    if (window.navbarManager) {
        window.navbarManager.updateCartCount();
    }

    // Show success message
    alert(`${currentProduct.name} berhasil ditambahkan ke keranjang!`);
}
```

```javascript
        alert('Toko tidak ditemukan!');
        window.location.href = 'homepage.html';
        return;
    }

    // Display store info
    displayStoreInfo();

    // Load store products
    await loadStoreProducts(storeId);

    // Check if current user is owner
    checkOwnership();
}

function displayStoreInfo() {
    document.getElementById('storeName').textContent = currentStore.name;
    document.getElementById('storeLocation').textContent = currentStore.location || 'Indonesia';
    document.getElementById('storeDescription').textContent = currentStore.description || 'Toko suku cadang terpercaya';

    if (currentStore.logo) {
        document.getElementById('storeLogo').src = currentStore.logo;
    }

    if (currentStore.banner) {
        document.getElementById('storeBanner').src = currentStore.banner;
    }

    // Display rating
    const rating = currentStore.rating || 0;
    document.getElementById('storeRating').textContent = rating.toFixed(1);
}

async function loadStoreProducts(storeId) {
    const response = await fetch('JSON/productData.json');
    const allProducts = await response.json();

    // Filter products by store ID
    storeProducts = allProducts.filter(p => p.storeId === storeId);

    // Update statistics
    document.getElementById('totalProducts').textContent = storeProducts.length;
    document.getElementById('totalSales').textContent = currentStore.totalSales || 0;
    document.getElementById('totalCustomers').textContent = currentStore.totalCustomers || 0;

    // Render products
    if (storeProducts.length === 0) {
        document.getElementById('noProducts').style.display = 'block';
        document.getElementById('storeProductsGrid').style.display = 'none';
    } else {
        renderProducts();
    }
}

function renderProducts() {
    const grid = document.getElementById('storeProductsGrid');
    grid.innerHTML = storeProducts.map(product => `
        <div class="product-card" onclick="viewProduct(${product.id})">
            <img src="${product.image}" alt="${product.name}">
            <div class="product-card-body">
                <h3>${product.name}</h3>
                ${product.discount ? `<span class="discount-badge">${product.discount}% OFF</span>` : ''}
                <p class="price">Rp ${product.price.toLocaleString()}</p>
            </div>
        </div>
    `).join('');
}

function viewProduct(productId) {
    window.location.href = `detail-produk.html?id=${productId}`;
}

function checkOwnership() {
    const user = JSON.parse(localStorage.getItem('loggedInUser'));

    if (user && user.storeId === currentStore.id) {
        // User is owner - show management buttons
        document.getElementById('editStoreBtn').style.display = 'block';
        document.getElementById('manageProductsBtn').style.display = 'block';
    }
}
```
